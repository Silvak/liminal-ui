# Accordion

An accordion component built with Ark UI.

<Callout type="info" title="Best practices">
  Use short titles in `AccordionTrigger` and clear content in `AccordionContent` for
  better scannability.
</Callout>

## Installation

```bash
liminal add accordion
```

## Usage

```tsx title="accordion-example.tsx" {10,17} showLineNumbers
import {
  Accordion,
  AccordionItem,
  AccordionTrigger,
  AccordionContent,
} from "@/components/ui/accordion";

export default function Example() {
  return (
    <Accordion defaultValue={["item-1"]} type="multiple">
      <AccordionItem value="item-1">
        <AccordionTrigger>Is it accessible?</AccordionTrigger>
        <AccordionContent>
          Yes. It adheres to the WAI-ARIA design pattern.
        </AccordionContent>
      </AccordionItem>
      <AccordionItem value="item-2">
        <AccordionTrigger>Is it styled?</AccordionTrigger>
        <AccordionContent>
          Yes. It uses Tailwind CSS tokens from your theme.
        </AccordionContent>
      </AccordionItem>
    </Accordion>
  );
}
```
