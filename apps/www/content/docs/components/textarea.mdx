# Textarea

A textarea component for multiline text input, visually aligned with the `Input` component.

<Preview>
  <Textarea placeholder="Write your message here" rows={4} />

```tsx title="textarea-basic.tsx" {3} showLineNumbers
import { Textarea } from "@/components/ui/textarea";

export default function Example() {
  return (
    <Textarea
      placeholder="Write your message here"
      rows={4}
    />
  );
}
```

</Preview>

## Installation

```bash
liminal add textarea
```

### Error state

Use `aria-invalid="true"` to show error styling, consistent with `Input`.

<Preview>
  <Textarea placeholder="Write your message here" aria-invalid="true" />

```tsx title="textarea-error.tsx" {3} showLineNumbers
import { Textarea } from "@/components/ui/textarea";

export default function Example() {
  return (
    <Textarea
      placeholder="Write your message here"
      aria-invalid="true"
    />
  );
}
```
