# Accordion

Componente acordeón construido con Ark UI. Usa `AccordionSimple` para listas tipo FAQ y la API compuesta para control total. Elige `type="single"` para un solo ítem abierto o `type="multiple"` para varios.

<Preview>
  <AccordionSimple
    type="multiple"
    defaultValue={["item-1"]}
    items={[
      {
        value: "item-1",
        title: "¿Es accesible?",
        content: "Sí. Sigue el patrón de diseño WAI-ARIA.",
      },
      {
        value: "item-2",
        title: "¿Tiene estilos?",
        content: "Sí. Usa los tokens de Tailwind CSS de tu tema.",
      },
    ]}
  />

```tsx title="accordion-rapido.tsx" showLineNumbers
import { AccordionSimple } from "@/components/ui/accordion";

export default function Example() {
  return (
    <AccordionSimple
      type="multiple"
      defaultValue={["item-1"]}
      items={[
        {
          value: "item-1",
          title: "¿Es accesible?",
          content: "Sí. Sigue el patrón de diseño WAI-ARIA.",
        },
        {
          value: "item-2",
          title: "¿Tiene estilos?",
          content: "Sí. Usa los tokens de Tailwind CSS de tu tema.",
        },
      ]}
    />
  );
}
```

</Preview>

<Callout type="info" title="Buenas prácticas">
  Usa títulos cortos en AccordionTrigger y contenido claro en AccordionContent para una mejor escaneabilidad.
</Callout>

## Instalación

```bash
liminal add accordion
```

### API compuesta (control total)

<Preview>
  <Accordion defaultValue={["item-1"]} type="multiple">
    <AccordionItem value="item-1">
      <AccordionTrigger>¿Es accesible?</AccordionTrigger>
      <AccordionContent>
        Sí. Sigue el patrón de diseño WAI-ARIA.
      </AccordionContent>
    </AccordionItem>
    <AccordionItem value="item-2">
      <AccordionTrigger>¿Tiene estilos?</AccordionTrigger>
      <AccordionContent>
        Sí. Usa los tokens de Tailwind CSS de tu tema.
      </AccordionContent>
    </AccordionItem>
  </Accordion>

```tsx title="accordion-compuesto.tsx" {10,17} showLineNumbers
import {
  Accordion,
  AccordionItem,
  AccordionTrigger,
  AccordionContent,
} from "@/components/ui/accordion";

export default function Example() {
  return (
    <Accordion defaultValue={["item-1"]} type="multiple">
      <AccordionItem value="item-1">
        <AccordionTrigger>¿Es accesible?</AccordionTrigger>
        <AccordionContent>
          Sí. Sigue el patrón de diseño WAI-ARIA.
        </AccordionContent>
      </AccordionItem>
      <AccordionItem value="item-2">
        <AccordionTrigger>¿Tiene estilos?</AccordionTrigger>
        <AccordionContent>
          Sí. Usa los tokens de Tailwind CSS de tu tema.
        </AccordionContent>
      </AccordionItem>
    </Accordion>
  );
}
```

</Preview>
