# Progress

A progress bar component built with Ark UI. Use the convenience API with `value` (and optional `showLabel`) or the compound API for custom layout.

<Preview>
  <Progress value={60} showLabel />

```tsx title="progress-convenience.tsx" showLineNumbers
import { Progress } from "@/components/ui/progress";

export default function Example() {
  return <Progress value={60} showLabel />;
}
```

</Preview>

## Installation

```bash
liminal add progress
```

### Compound API

Use `ProgressRoot`, `ProgressLabel`, `ProgressValueText`, `ProgressTrack`, and `ProgressRange` for custom layout.

```tsx title="progress-compound.tsx" showLineNumbers
import {
  ProgressRoot,
  ProgressLabel,
  ProgressValueText,
  ProgressTrack,
  ProgressRange,
} from "@/components/ui/progress";

export default function Example() {
  return (
    <ProgressRoot value={50} min={0} max={100}>
      <div className="mb-2 flex justify-between">
        <ProgressLabel>Upload</ProgressLabel>
        <ProgressValueText />
      </div>
      <ProgressTrack>
        <ProgressRange />
      </ProgressTrack>
    </ProgressRoot>
  );
}
```
